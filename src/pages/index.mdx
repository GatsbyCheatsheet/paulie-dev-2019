  {(tags) => (
    <Box>
      {tags
        .sort((a, b) => b.percent - a.percent)
        .map((tag, index) => {
          const { name, count, percent } = tag;
          return (
            <Badge
              key={index}
              variant="primary"
              sx={{
                mr: 2,
                mb: 2,
                color: mix("secondary", "primary", `${index / tags.length}`),
                borderColor: mix(
                  "secondary",
                  "primary",
                  `${index / tags.length}`
                ),
              }}
            >
              {name}
            </Badge>
          );
        })}
    </Box>
  )}
</SourceTags>

## latest posts

<Divider />

<SourceList filter="posts">
  {(posts) => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: (theme) => `-${theme.space[2]}px`,
        mr: (theme) => `-${theme.space[2]}px`,
      }}
    >
      {posts
        .sort(
          (a, b) =>
            new Date(b.node.frontmatter.date) -
            new Date(a.node.frontmatter.date)
        )
        .splice(0, 2)
        .map((edge, index) => {
          const {
            frontmatter: { title, featuredImage, tags, date },
            excerpt,
            fields: { slug },
          } = edge.node;
          return (
            <PostCard
              key={index}
              title={title}
              featuredImage={featuredImage}
              tags={tags}
              date={date}
              excerpt={excerpt}
              slug={slug}
            />
          );
        })}
    </Flex>
  )}
</SourceList>
